001package forest;
002
003import java.awt.Dimension;
004import java.io.File;
005import java.io.FileNotFoundException;
006
007import javax.swing.JFileChooser;
008import javax.swing.JFrame;
009import javax.swing.filechooser.FileFilter;
010import javax.swing.filechooser.FileNameExtensionFilter;
011
012/**
013 * テキストファイルを読み取り樹状整列を実行するクラス。<br>
014 * Makefileを用いた実行方法は以下の通りです。<br>
015 * $ make test<br>
016 * その後にファイル選択ウィンドウをから表示したいファイル選択してください<br>
017 * @author Takumi Koike 744438
018 */
019public class Example extends Object
020{
021        /**
022         * ファイルチューザーでファイルを選択し、樹上整列を表示する準備をする
023         * @param arguments 引数はなく実行できるので不要
024         * @throws FileNotFoundException 指定されたパス名で示されるファイルが開けなかったことを通知
025         */
026        public static void main(String[] arguments) throws FileNotFoundException
027        {
028                JFileChooser chooser = new JFileChooser();
029
030                String currentDirectory = System.getProperty("user.dir");
031                //String dataDirectory = currentDirectory +  File.separator + "src" + File.separator + "forest" + File.separator + "tree.txt";
032                String dataDirectory = currentDirectory +  File.separator + "resource" + File.separator + "data" + File.separator + "tree.txt";
033
034                //デフォルトの選択ファイルを指定
035                chooser.setSelectedFile(new File(dataDirectory));
036
037                FileFilter filter = new FileNameExtensionFilter("textファイル", "txt");
038                chooser.addChoosableFileFilter(filter);
039                chooser.setAcceptAllFileFilterUsed(false);
040
041                chooser.setDialogTitle("ファイルを読み込む");
042
043                //ダイアログを表示
044                int selected = chooser.showOpenDialog(null);
045                File aFile;
046                if(selected == JFileChooser.APPROVE_OPTION){
047                        aFile = chooser.getSelectedFile();
048
049                        // MVCを作成する。
050                        ForestModel aModel = new ForestModel(aFile);
051                        ForestView aView = new ForestView(aModel);
052
053                        // ウィンドウを生成して開く。
054                        JFrame aWindow = new JFrame(aFile.getName());
055                        aWindow.getContentPane().add(aView);
056                        aWindow.setMinimumSize(new Dimension(400, 300));
057                        aWindow.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
058                        aWindow.setSize(800, 600);
059                        aWindow.setLocationRelativeTo(null);
060                        aWindow.setVisible(true);
061
062                        // 樹状整列のアニメーションを行う。
063                        aModel.animate();
064        }
065
066                return;
067        }
068}
